# Plan 1.1: Invite DB Model + Server Action

## Objective
Add the `Invite` database model and a `sendInvites` server action so the organizer can invite households to an event by email. Each invite gets a unique token and a SEND_INVITE job enqueued.

## Context
- prisma/schema.prisma
- src/app/admin/actions.ts

## Tasks

<task type="auto">
  <name>Add Invite model to Prisma schema and migrate</name>
  <files>
    prisma/schema.prisma
  </files>
  <action>
    Add the following model to `prisma/schema.prisma` after the `OtpCode` model:

    ```prisma
    model Invite {
      id        String   @id @default(uuid())
      eventId   String
      event     Event    @relation(fields: [eventId], references: [id], onDelete: Cascade)
      email     String
      token     String   @unique @default(uuid())
      status    String   @default("PENDING") // PENDING, ACCEPTED, DECLINED
      createdAt DateTime @default(now())

      @@unique([eventId, email])
    }
    ```

    Also add the relation back-reference on the `Event` model:
    ```prisma
    invites  Invite[]
    ```
    Place it after the existing `matches  Match[]` line on the Event model.

    Then run:
    ```
    npx prisma migrate dev --name add_invite
    ```

    Add a new exported server action `sendInvites` to the bottom of `src/app/admin/actions.ts`.

    ```ts
    export async function sendInvites(formData: FormData) {
        const { email: organizer } = await requireSession();

        const eventId = formData.get('eventId') as string | null;
        if (!eventId) return { error: 'Missing event ID.' };

        const event = await db.event.findUnique({ where: { id: eventId } });
        if (!event) return { error: 'Event not found.' };
        if (event.createdByEmail !== organizer) return { error: 'Only the event creator can send invites.' };

        const rawEmails = (formData.get('emails') as string | null) ?? '';
        const emails = rawEmails
            .split(/[\n,]+/)
            .map(e => e.trim().toLowerCase())
            .filter(e => e.length > 0);

        if (emails.length === 0) return { error: 'Enter at least one email address.' };

        let sent = 0;
        for (const email of emails) {
            const existing = await db.invite.findUnique({
                where: { eventId_email: { eventId, email } },
            });
            if (existing) continue; // skip duplicates silently

            const invite = await db.invite.create({
                data: { eventId, email },
            });

            await db.job.create({
                data: {
                    type: 'SEND_INVITE',
                    payload: JSON.stringify({
                        email,
                        eventId,
                        eventName: event.name,
                        inviteToken: invite.token,
                    }),
                },
            });
            sent++;
        }

        revalidatePath('/admin');
        return { success: true, sent };
    }
    ```

    No schema changes needed here — Prisma client is regenerated by the migration in the previous task.
  </action>
